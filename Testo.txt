<script>
// ══════════════════════════════
//  STORAGE
// ══════════════════════════════
const STORE_KEY  = 'canzoniere_songs';
const PWD_KEY    = 'canzoniere_pwd';
const DEFAULT_PWD = '1234';

function getHash(s) {
  // semplice hash djb2
  let h = 5381;
  for (let i = 0; i < s.length; i++) h = ((h << 5) + h) ^ s.charCodeAt(i);
  return (h >>> 0).toString(36);
}
function getPwd() { return localStorage.getItem(PWD_KEY) || getHash(DEFAULT_PWD); }
function getSongs() { try { return JSON.parse(localStorage.getItem(STORE_KEY)) || []; } catch { return []; } }
function setSongs(arr) { localStorage.setItem(STORE_KEY, JSON.stringify(arr)); }

// ══════════════════════════════
//  LOCK
// ══════════════════════════════
document.getElementById('pwdInput').addEventListener('keydown', e => { if (e.key === 'Enter') unlock(); });

function unlock() {
  const val = document.getElementById('pwdInput').value;
  const stored = getPwd();
  const isDefault = stored === getHash(DEFAULT_PWD) && val === DEFAULT_PWD;
  const isMatch   = stored === getHash(val);
  if (isDefault || isMatch) {
    document.getElementById('lockScreen').style.display = 'none';
    document.getElementById('app').style.display = 'flex';
    renderList();
  } else {
    const el = document.getElementById('lockError');
    el.textContent = 'Password errata.';
    document.getElementById('pwdInput').value = '';
    setTimeout(() => el.textContent = '', 2000);
  }
}

// ══════════════════════════════
//  LISTA
// ══════════════════════════════
let currentId = null;

function renderList() {
  const q = document.getElementById('searchInput').value.toLowerCase();
  const songs = getSongs().filter(s =>
    s.title.toLowerCase().includes(q) || (s.artist||'').toLowerCase().includes(q)
  );
  const sidebar = document.getElementById('sidebar');
  document.getElementById('countBadge').textContent = getSongs().length + ' canzon' + (getSongs().length === 1 ? 'e' : 'i');
  if (!songs.length) {
    sidebar.innerHTML = `<div class="empty-list">Nessuna canzone trovata.</div>`;
    return;
  }
  sidebar.innerHTML = songs.map(s => `
    <div class="song-item ${s.id === currentId ? 'active' : ''}" onclick="selectSong('${s.id}')">
      <div class="st">${esc(s.title)}</div>
      ${s.artist ? `<div class="sa">${esc(s.artist)}</div>` : ''}
    </div>
  `).join('');
}

function selectSong(id) {
  currentId = id;
  renderList();
  const s = getSongs().find(x => x.id === id);
  if (!s) return;
  const main = document.getElementById('main');
  main.innerHTML = `
    <div class="panel-meta">
      <div class="panel-title">${esc(s.title)}</div>
      ${s.artist ? `<div class="panel-artist">${esc(s.artist)}</div>` : ''}
    </div>
    <div class="panel-bar">
      <button class="btn sm" onclick="openModal('${s.id}')">✏ Modifica</button>
      <button class="btn danger sm" onclick="deleteSong('${s.id}')">🗑 Elimina</button>
    </div>
    <div class="lyrics-box">${esc(s.lyrics)}</div>
  `;
}

function deleteSong(id) {
  if (!confirm('Eliminare questa canzone?')) return;
  const songs = getSongs().filter(s => s.id !== id);
  setSongs(songs);
  currentId = null;
  document.getElementById('main').innerHTML = `<div class="welcome" id="welcome"><div class="welcome-icon">♫</div><p>Seleziona una canzone dalla lista.</p></div>`;
  renderList();
  showToast('Canzone eliminata.');
}

// ══════════════════════════════
//  MODAL AGGIUNGI / MODIFICA
// ══════════════════════════════
let editingId = null;

function openModal(id) {
  editingId = id || null;
  document.getElementById('modalTitle').textContent = id ? 'Modifica canzone' : 'Nuova canzone';
  clearFileStatus();
  if (id) {
    const s = getSongs().find(x => x.id === id);
    document.getElementById('fTitle').value  = s.title;
    document.getElementById('fArtist').value = s.artist || '';
    document.getElementById('fLyrics').value = s.lyrics;
  } else {
    document.getElementById('fTitle').value  = '';
    document.getElementById('fArtist').value = '';
    document.getElementById('fLyrics').value = '';
  }
  document.getElementById('fileInput').value = '';
  document.getElementById('modal').classList.add('open');
}

function closeModal() {
  document.getElementById('modal').classList.remove('open');
  editingId = null;
}

function saveSong() {
  const title  = document.getElementById('fTitle').value.trim();
  const artist = document.getElementById('fArtist').value.trim();
  const lyrics = document.getElementById('fLyrics').value.trim();
  if (!title) { alert('Il titolo è obbligatorio.'); return; }
  const songs = getSongs();
  if (editingId) {
    const idx = songs.findIndex(s => s.id === editingId);
    if (idx !== -1) { songs[idx].title = title; songs[idx].artist = artist; songs[idx].lyrics = lyrics; }
  } else {
    songs.push({ id: Date.now().toString(36) + Math.random().toString(36).slice(2,5), title, artist, lyrics });
  }
  setSongs(songs);
  closeModal();
  renderList();
  if (editingId || currentId) selectSong(editingId || currentId);
  showToast(editingId ? 'Canzone aggiornata.' : 'Canzone aggiunta!');
}

function dzDragOver(e)  { e.preventDefault(); document.getElementById('dropZone').classList.add('dragover'); }
function dzDragLeave(e) { document.getElementById('dropZone').classList.remove('dragover'); }
function dzDrop(e) {
  e.preventDefault();
  document.getElementById('dropZone').classList.remove('dragover');
  const file = e.dataTransfer.files[0];
  if (file) handleFile(file);
}

function setFileStatus(type, msg) {
  const el = document.getElementById('fileStatus');
  el.className = 'file-status ' + type;
  el.textContent = msg;
}
function clearFileStatus() {
  const el = document.getElementById('fileStatus');
  el.className = 'file-status';
  el.textContent = '';
}

async function handleFile(file) {
  if (!file) return;
  const name = file.name.toLowerCase();
  setFileStatus('loading', '⏳ Lettura file in corso…');

  // Auto-fill titolo se il campo è vuoto
  if (!document.getElementById('fTitle').value.trim()) {
    let base = file.name.replace(/\.[^.]+$/, '').replace(/[-_]/g, ' ');
    document.getElementById('fTitle').value = base;
  }

  try {
    if (name.endsWith('.txt')) {
      const text = await file.text();
      document.getElementById('fLyrics').value = text;
      setFileStatus('ok', '✓ File .txt caricato correttamente.');
    }
    else if (name.endsWith('.pdf')) {
      await readPDF(file);
    }
    else if (name.endsWith('.docx')) {
      await readDOCX(file);
    }
    else {
      setFileStatus('err', '⚠ Formato non supportato. Usa .txt, .pdf o .docx.');
    }
  } catch(err) {
    setFileStatus('err', '⚠ Errore nella lettura: ' + err.message);
  }
}

async function readPDF(file) {
  const pdfjsLib = window['pdfjs-dist/build/pdf'];
  pdfjsLib.GlobalWorkerOptions.workerSrc = 'https://cdnjs.cloudflare.com/ajax/libs/pdf.js/3.11.174/pdf.worker.min.js';
  const ab = await file.arrayBuffer();
  const pdf = await pdfjsLib.getDocument({ data: ab }).promise;
  let fullText = '';
  for (let p = 1; p <= pdf.numPages; p++) {
    const page = await pdf.getPage(p);
    const content = await page.getTextContent();
    const pageText = content.items.map(i => i.str).join(' ');
    fullText += pageText + '\n';
  }
  document.getElementById('fLyrics').value = fullText.trim();
  setFileStatus('ok', `✓ PDF letto (${pdf.numPages} pagina/e).`);
}

async function readDOCX(file) {
  const ab = await file.arrayBuffer();
  const result = await mammoth.extractRawText({ arrayBuffer: ab });
  document.getElementById('fLyrics').value = result.value.trim();
  setFileStatus('ok', '✓ File .docx caricato correttamente.');
}

function openSettings() { document.getElementById('settingsModal').classList.add('open'); }
function closeSettings() {
  document.getElementById('settingsModal').classList.remove('open');
  ['sPwdOld','sPwdNew','sPwdConf'].forEach(id => document.getElementById(id).value = '');
  document.getElementById('pwdMsg').textContent = '';
}

function changePassword() {
  const old  = document.getElementById('sPwdOld').value;
  const nw   = document.getElementById('sPwdNew').value;
  const conf = document.getElementById('sPwdConf').value;
  const msg  = document.getElementById('pwdMsg');

  const stored  = getPwd();
  const isDefault = stored === getHash(DEFAULT_PWD) && old === DEFAULT_PWD;
  if (!isDefault && stored !== getHash(old)) {
    msg.style.color = 'var(--danger)'; msg.textContent = 'Password attuale errata.'; return;
  }
  if (nw.length < 3) { msg.style.color = 'var(--danger)'; msg.textContent = 'Nuova password troppo corta (min 3 caratteri).'; return; }
  if (nw !== conf)   { msg.style.color = 'var(--danger)'; msg.textContent = 'Le due password non coincidono.'; return; }
  localStorage.setItem(PWD_KEY, getHash(nw));
  msg.style.color = 'var(--success)'; msg.textContent = '✓ Password cambiata!';
  setTimeout(closeSettings, 1500);
}

function exportData() {
  const data = JSON.stringify({ songs: getSongs() }, null, 2);
  const blob = new Blob([data], { type: 'application/json' });
  const a = document.createElement('a');
  a.href = URL.createObjectURL(blob);
  a.download = 'canzoniere_backup.json';
  a.click();
}

function importData(e) {
  const file = e.target.files[0];
  if (!file) return;
  const reader = new FileReader();
  reader.onload = evt => {
    try {
      const obj = JSON.parse(evt.target.result);
      if (!Array.isArray(obj.songs)) throw new Error('Formato non valido.');
      const current = getSongs();
      const merged = [...current];
      obj.songs.forEach(s => {
        if (!merged.find(x => x.id === s.id)) merged.push(s);
      });
      setSongs(merged);
      renderList();
      showToast(`Importazione completata: ${obj.songs.length} canzoni.`);
      closeSettings();
    } catch(err) {
      alert('Errore importazione: ' + err.message);
    }
  };
  reader.readAsText(file);
}

function esc(s) {
  return String(s||'')
    .replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;')
    .replace(/"/g,'&quot;');
}

let toastTimer;
function showToast(msg) {
  const t = document.getElementById('toast');
  t.textContent = msg;
  t.classList.add('show');
  clearTimeout(toastTimer);
  toastTimer = setTimeout(() => t.classList.remove('show'), 2500);
}
